Multi Thread
```
import datetime
from threading import Thread

def intruder(i):
	for a in range(10):
		print "Thread #{} -> {} -  \r\n".format(i,a,datetime.datetime.now())

for i in range(10):
	t=Thread(target=intruder,args=(i,))
	t.start()
```
